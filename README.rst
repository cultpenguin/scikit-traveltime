.. code:: 

   import numpy as np
   import traveltime as tt
   
   #%% CREATE REFERENCE VELOCITY MODEL
   dx=0.1;
   x = np.arange(-1,6,dx)
   y = np.arange(-1,13,dx)
   V=0.1*np.ones((len(y),len(x)))
   
   #%% SET SOURCE AND RECEIVERS
   S=np.array([[0,2],[0,5],[0,4]])
   R=np.array([[5,12],[5,5],[5,1]])
   
   t = tt.eikonal_traveltime(x,y,[],V,S,R)
